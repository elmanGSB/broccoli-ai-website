> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the 4-step `/demo` lead-capture and consultation-scheduling flow for broccolli.ai.

**Architecture:** Single `src/pages/demo.astro` page; vanilla JS show/hide step manager (no page reloads). Step 1 dual-posts to Formspree (email alert) and jumpersapp.com/api/leads (FastAPI on Paperclip VM, reverse-proxied via Caddy, writes to Postgres). Step 2 PATCHes jumpersapp.com/api/leads to record pillar. Steps 3–4 are client-only. HTTPS enforced throughout.

**Tech Stack:** Astro 6 (static), vanilla JS, Formspree, FastAPI + asyncpg on Paperclip VM, Caddy reverse proxy on jumpersapp.com, PostgreSQL (discovery schema), Playwright e2e

## Task 2: VM Leads API

- [ ] **Step 1: Write the FastAPI service**

```python
# discovery/leads_service.py
"""
Leads API — receives demo form submissions from broccolli.ai/demo.
Runs on Paperclip VM port 3101.

Deploy:
  scp leads_service.py paperclip-vm:~/leads_service.py
  ssh paperclip-vm "sudo systemctl restart leads-api"
"""

import os
from contextlib import asynccontextmanager

import asyncpg
from fastapi import FastAPI, HTTPException
from fastapi.middleware.cors import CORSMiddleware
from pydantic import BaseModel, EmailStr

DATABASE_URL = os.environ.get(
    "DATABASE_URL",
    "postgresql://paperclip:paperclip@127.0.0.1:5432/discovery",
)

pool: asyncpg.Pool | None = None


@asynccontextmanager
async def lifespan(app: FastAPI):
    global pool
    pool = await asyncpg.create_pool(DATABASE_URL, min_size=1, max_size=5)
    yield
    await pool.close()


app = FastAPI(lifespan=lifespan)

app.add_middleware(
    CORSMiddleware,
    allow_origins=["https://broccolli.ai", "http://localhost:4321"],
    allow_methods=["POST", "PATCH", "OPTIONS"],
    allow_headers=["Content-Type"],
)


class LeadCreate(BaseModel):
    name: str
    email: EmailStr
    phone: str = ""
    company: str = ""
    company_size: str = ""


class LeadPatch(BaseModel):
    pillar: str


@app.post("/api/leads", status_code=201)
async def create_lead(body: LeadCreate):
    row = await pool.fetchrow(
        """
        INSERT INTO discovery.leads (name, email, phone, company, company_size)
        VALUES ($1, $2, $3, $4, $5)
        RETURNING id
        """,
        body.name,
        body.email,
        body.phone,
        body.company,
        body.company_size,
    )
    return {"id": row["id"]}


@app.patch("/api/leads/{lead_id}", status_code=200)
async def update_lead_pillar(lead_id: int, body: LeadPatch):
    result = await pool.execute(
        "UPDATE discovery.leads SET pillar = $1 WHERE id = $2",
        body.pillar,
        lead_id,
    )
    if result == "UPDATE 0":
        raise HTTPException(status_code=404, detail="Lead not found")
    return {"ok": True}


@app.get("/health")
async def health():
    return {"status": "ok"}
```

var _leadId = null;

/* ── Step manager ── */
function showStep(n) {
  [1, 2, 3, 4].forEach(function(i) {
    var el = document.getElementById('step' + i);
    if (el) el.style.display = (i === n) ? 'flex' : 'none';

/* ── Pillar selection ── */
function pick(el) {
  document.querySelectorAll('.pillar-card, .custom-card').forEach(function(c) { c.classList.remove('selected'); });
  el.classList.add('selected');
  _selectedPillar = el.dataset.id;

function pickCustom(el) {
  document.querySelectorAll('.pillar-card, .custom-card').forEach(function(c) { c.classList.remove('selected'); });
  el.classList.add('selected');
  _selectedPillar = 'custom';

/* ── Validation ── */
function setErr(id, errId, msg) {
  var el = document.getElementById(id);
  var err = document.getElementById(errId);
  el.classList.add('error'); el.classList.remove('valid');
  err.textContent = '↑ ' + msg; err.classList.add('visible');
}
function clearErr(id, errId) {
  var el = document.getElementById(id);
  var err = document.getElementById(errId);
  el.classList.remove('error'); el.classList.add('valid');
  err.classList.remove('visible');
}
function validateField(id) {
  var el = document.getElementById(id);
  var v = el.value.trim();
  switch (id) {
    case 'name':
      if (!v || v.length < 2) { setErr('name', 'name-err', 'Please enter your full name.'); return false; }
      clearErr('name', 'name-err'); break;
    case 'company':
      if (!v || v.length < 2) { setErr('company', 'company-err', 'Please enter your company name.'); return false; }
      clearErr('company', 'company-err'); break;
    case 'email':
      if (!v) { setErr('email', 'email-err', 'Work email is required.'); return false; }
      if (!/^[^\s@]+@[^\s@]+\.[^\s@]{2,}$/.test(v)) { setErr('email', 'email-err', 'Please enter a valid email address.'); return false; }
      clearErr('email', 'email-err'); break;
    case 'phone':
      if (!v) { setErr('phone', 'phone-err', 'Phone number is required.'); return false; }
      if (v.replace(/\D/g, '').length < 7) { setErr('phone', 'phone-err', 'Please enter a valid phone number.'); return false; }
      clearErr('phone', 'phone-err'); break;
    case 'size':
      if (!el.value) { setErr('size', 'size-err', 'Please select your company size.'); return false; }
      clearErr('size', 'size-err'); break;
  }
  return true;
}

['name', 'company', 'email', 'phone', 'size'].forEach(function(id) {
  var el = document.getElementById(id);
  if (!el) return;
  el.addEventListener('blur', function() { validateField(id); });
  el.addEventListener('input', function() { if (el.classList.contains('error')) validateField(id); });
});

/* ── Step 1 submit ── */
async function submitStep1() {
  var fields = ['name', 'company', 'email', 'phone', 'size'];
  var allValid = fields.every(function(id) { return validateField(id); });
  if (!allValid) {
    var btn = document.getElementById('step1-submit');
    btn.classList.remove('shake'); void btn.offsetWidth; btn.classList.add('shake');
    return;
  }

  var btn = document.getElementById('step1-submit');
  btn.disabled = true;
  btn.textContent = 'Sending…';

  var payload = {
    name: document.getElementById('name').value.trim(),
    email: document.getElementById('email').value.trim(),
    phone: document.getElementById('phone').value.trim(),
    company: document.getElementById('company').value.trim(),
    company_size: document.getElementById('size').value,
  };

  try {
    // Dual-post: Formspree (email alert) + VM API (Postgres)
    var [formspreeRes, vmRes] = await Promise.allSettled([
      fetch(FORMSPREE_ENDPOINT, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json', 'Accept': 'application/json' },
        body: JSON.stringify(payload),
      }),
      fetch(LEADS_API + '/api/leads', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(payload),
      }),
    ]);

    // Save lead id for pillar PATCH (non-blocking if VM is down)
    if (vmRes.status === 'fulfilled' && vmRes.value.ok) {
      var data = await vmRes.value.json();
      _leadId = data.id;
    }
  } catch (e) {
    // Non-fatal: don't block the user if either service is down
    console.warn('Lead submission error:', e);
  }

  btn.disabled = false;
  btn.textContent = 'Continue →';
  showStep(2);
}

/* ── Step 2 → 3: PATCH pillar ── */
async function goToStep3() {
  if (_leadId && _selectedPillar) {
    fetch(LEADS_API + '/api/leads/' + _leadId, {
      method: 'PATCH',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ pillar: _selectedPillar }),
    }).catch(function() {}); // fire-and-forget
  }
  showStep(3);
}

/* ── Step 3 → 4 ── */
function goToStep4() {
  var labels = {
    sales: '01 · Sales Empowerment',

    payments: '04 · Payments & Collections',

  };
  var label = document.getElementById('selected-pillar-label');
  if (label && _selectedPillar && labels[_selectedPillar]) {
    label.textContent = labels[_selectedPillar];
  }
  showStep(4);
  // Trigger checkmark animation
  var mark = document.getElementById('confirm-mark');
  if (mark) { mark.classList.remove('animate'); void mark.offsetWidth; mark.classList.add('animate'); }
}
